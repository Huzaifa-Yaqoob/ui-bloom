## Usage

```tsx
import { PasswordInput } from "@/components/ui/password-input";
```

```tsx
<FormField
  control={form.control}
  name="password"
  render={({ field }) => (
    <FieldRenderer label="Password">
      <PasswordInput placeholder="password" {...field} />
    </FieldRenderer>
  )}
/>
```

## Dependencies
